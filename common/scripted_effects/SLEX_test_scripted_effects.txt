test_add_resources = {
    owner = { add_resource = { energy = $VALUE$ } }
    SLEX_dump_param = {
        EFFECT = $EFFECT|null$
        TYPE = $TYPE|null$
        INDEX = $INDEX|null$
        VALUE = $VALUE|null$
        LIST = $LIST|null$
        FCT = $FCT|null$
    }
}

test_effect = {
    if = {
        limit = {
            check_variable = {
                which = $LIST$_$INDEX$
                value > $VALUE$
            }
        }
        owner = { add_resource = { energy = 100 } }
    }
}

test_check_list_trigger = {
    if = {
        limit = {
            SLEX_check_list_variable_const_index = { LIST = $LIST$ INDEX = $INDEX$ OPERATOR = $OPERATOR$ VALUE = $VALUE$ }
        }
        owner = { add_resource = { energy = 100 } }
    }
}

test_custom_trigger = {
    SLEX_apply_effect_conditionnal_list_variable_const_index = { TRIGGER = SLEX_test_trigger EFFECT = test_add_resources LIST = $LIST$ INDEX = $INDEX$ VALUE = $VALUE$}
}
